결론
오늘 3월 4일 장은 뉴욕증시 추가 하락과 환율 1,500원 돌파 충격으로 추가 하락 출발이 유력합니다. 그러나 전일 7.24% 급락이라는 패닉셀이 이미 발생한 만큼, 낙폭은 점차 축소될 가능성이 높습니다.

가장 중요한 판단 기준은 **"전쟁이 1~2주 내 진정되느냐, 장기화되느냐"**입니다. 과거 중동전쟁 사례와 다수 증권사의 분석을 종합하면, 전쟁이 1주 내 소강 국면에 접어들 경우 현재 수준은 중장기 매수의 절호의 기회가 될 수 있습니다. 반면 확전 시에는 코스피 5,000대 초반까지 열어둬야 하므로, 분할매수와 현금 확보를 반드시 병행하시기 바랍니다.

이전 이란 공습 후 한국 증시 변화패턴, 어제의 한국증시 반응, 앞으로의 변화를 예측하여, 지수, 주도섹터 최우선 대응종목을 대상으로 하는 멀티차트 기반 당일+중기 자동매매 전략을 만들어주세요

# CHART 1: KOSPI INDEX MACRO ENGINE
# 코스피 지수 + VIX + 환율 + 유가 연동 매크로 판단
# ============================================================

import numpy as np
import pandas as pd
from datetime import datetime, time

class MacroRegimeEngine:
    """
    시장 국면(Regime)을 실시간 판단하여
    나머지 3개 차트의 공격성(베타 배수)을 동적 조절
    """
    
    # ── 과거 중동전쟁 패턴 기반 파라미터 ──
    WAR_PATTERN = {
        'avg_initial_drop': -0.07,      # 3/3일 -7.24% (이번 실제값)
        'avg_1week_recovery': +0.031,    # 과거 평균 1주후 +3.1%
        'avg_1month_recovery': +0.046,   # 과거 평균 1개월후 +4.6%
        'max_drawdown_scenario': -0.15,  # LS증권 최악 시나리오 -15%
        'recovery_start_day': 2,         # 통상 2~3거래일 후 반등 시작
        'full_recovery_days': 20,        # 평균 20거래일(1개월) 내 완전회복
    }
    
    # ── 핵심 임계값 ──
    THRESHOLDS = {
        'kospi_crash_level': 5500,       # 극단 하락 시 코스피 지지선
        'kospi_recovery_target': 6200,   # 회복 목표 (전고점)
        'usdkrw_danger': 1500,           # 환율 위험선
        'usdkrw_safe': 1440,             # 환율 안전선
        'wti_danger': 90,                # 유가 위험선 (장기 인플레 우려)
        'wti_moderate': 80,              # 유가 경계선
        'vkospi_panic': 35,              # VKOSPI 공포 수준
        'vkospi_normal': 20,             # VKOSPI 정상 수준
    }
    
    def __init__(self):
        self.current_regime = 'CRISIS'   # 초기: 위기 모드
        self.war_day_count = 4           # 전쟁 4일차 (2/28 공습 기준)
        self.beta_multiplier = 0.5       # 초기 공격성 50%
        
    def determine_regime(self, kospi, usdkrw, wti, vkospi, 
                         foreign_net_buy, individual_net_buy):
        """
        5단계 시장 국면 판단
        Returns: regime_name, beta_multiplier (0.0 ~ 2.0)
        """
        
        score = 0  # -10(극단위기) ~ +10(극단강세)
        
        # ── Factor 1: 코스피 위치 (과거 패턴 대비) ──
        drop_from_high = (kospi - 6244) / 6244  # 3/2 종가 6244 기준
        expected_drop = self.WAR_PATTERN['avg_initial_drop']
        
        if drop_from_high < expected_drop * 1.5:   # 과거 평균 1.5배 이상 하락
            score -= 3  # 과매도
        elif drop_from_high < expected_drop:        # 과거 평균 수준 하락
            score -= 1
        elif drop_from_high > -0.03:                # 3% 미만 하락 (회복 중)
            score += 2
            
        # ── Factor 2: 환율 스트레스 ──
        if usdkrw > self.THRESHOLDS['usdkrw_danger']:
            score -= 2
        elif usdkrw < self.THRESHOLDS['usdkrw_safe']:
            score += 1
            
        # ── Factor 3: 유가 리스크 ──
        if wti > self.THRESHOLDS['wti_danger']:
            score -= 3  # 인플레이션 공포
        elif wti > self.THRESHOLDS['wti_moderate']:
            score -= 1
        else:
            score += 1
            
        # ── Factor 4: 변동성(VKOSPI) ──
        if vkospi > self.THRESHOLDS['vkospi_panic']:
            score -= 2
        elif vkospi < self.THRESHOLDS['vkospi_normal']:
            score += 2
            
        # ── Factor 5: 수급 (핵심 변수) ──
        # 외국인 -5조 이상이면 위기, 개인 순매수가 외국인 매도 상쇄하면 지지
        net_supply = foreign_net_buy + individual_net_buy  # 억원 단위
        if net_supply > 10000:     # 1조 이상 순유입
            score += 2
        elif net_supply > 0:
            score += 1
        elif foreign_net_buy < -30000:  # 외국인 3조 이상 매도
            score -= 2
            
        # ── Factor 6: 전쟁 일수 패턴 (과거 데이터 기반) ──
        if self.war_day_count <= 2:
            score -= 2  # 초기 충격기
        elif self.war_day_count <= 5:
            score -= 1  # 불확실성 지속
        elif self.war_day_count <= 10:
            score += 1  # 반등 시작 구간 (과거 패턴)
        else:
            score += 2  # 정상화 구간
            
        # ── Regime 결정 ──
        if score <= -6:
            self.current_regime = 'EXTREME_CRISIS'
            self.beta_multiplier = 0.2   # 극소 포지션만
        elif score <= -3:
            self.current_regime = 'CRISIS'
            self.beta_multiplier = 0.5   # 반토막 포지션
        elif score <= 0:
            self.current_regime = 'CAUTIOUS'
            self.beta_multiplier = 0.7   # 신중 모드
        elif score <= 4:
            self.current_regime = 'RECOVERY'
            self.beta_multiplier = 1.0   # 정상 포지션
        else:
            self.current_regime = 'AGGRESSIVE'
            self.beta_multiplier = 1.5   # 공격적 매수
            
        return self.current_regime, self.beta_multiplier
    
    def get_war_phase_signal(self):
        """
        전쟁 진행 단계별 전략 시그널
        과거 6건의 중동 전쟁 패턴 기반
        """
        if self.war_day_count <= 1:
            return {
                'phase': 'SHOCK',
                'action': 'HOLD_CASH',
                'defense_sector_weight': 0.6,   # 방산/정유 비중 60%
                'growth_sector_weight': 0.1,    # 반도체 비중 10%
                'cash_weight': 0.3,             # 현금 30%
            }
        elif self.war_day_count <= 3:
            return {
                'phase': 'PANIC_SELLING',
                'action': 'SELECTIVE_BUY',
                'defense_sector_weight': 0.5,
                'growth_sector_weight': 0.2,    # 반도체 분할매수 시작
                'cash_weight': 0.3,
            }
        elif self.war_day_count <= 7:
            return {
                'phase': 'STABILIZATION',       # ★ 현재 위치 (4일차)
                'action': 'ACCUMULATE',
                'defense_sector_weight': 0.4,
                'growth_sector_weight': 0.35,   # 반도체 본격 매수
                'cash_weight': 0.25,
            }
        elif self.war_day_count <= 14:
            return {
                'phase': 'EARLY_RECOVERY',
                'action': 'AGGRESSIVE_BUY',
                'defense_sector_weight': 0.3,
                'growth_sector_weight': 0.5,    # 반도체 주도
                'cash_weight': 0.2,
            }
        else:
            return {
                'phase': 'FULL_RECOVERY',
                'action': 'RIDE_MOMENTUM',
                'defense_sector_weight': 0.2,
                'growth_sector_weight': 0.6,
                'cash_weight': 0.2,
            }

# CHART 2: DEFENSE SECTOR — SCALPING + SWING ENGINE
# 대상: 한화에어로(012450), LIG넥스원(079550), 한화시스템(272210),

class DefenseSectorEngine:
    """
    방산주 전용 멀티타임프레임 전략
    - 당일: 1분/5분봉 VWAP+RSI 스캘핑
    - 중기: 일봉 볼린저밴드+MACD 스윙
    """
    
    TARGETS = {
        '012450': {  # 한화에어로스페이스
            'name': '한화에어로스페이스',
            'prev_close': 1432000,
            'target_price_consensus': 1800000,   # 한투증권 목표가 180만
            'stop_loss_pct': -0.08,
            'position_weight': 0.35,             # 방산 내 비중 35%
            'avg_daily_volume': 2500000,
            'war_premium_factor': 1.3,           # 전쟁 프리미엄 30%
        },
        '079550': {  # LIG넥스원
            'name': 'LIG넥스원',
            'prev_close': 661000,
            'target_price_consensus': 710000,    # 하나증권 목표가 71만
            'stop_loss_pct': -0.08,
            'position_weight': 0.30,
            'avg_daily_volume': 800000,
            'war_premium_factor': 1.4,           # 천궁 실전검증 프리미엄
        },
        '272210': {  # 한화시스템
            'name': '한화시스템',
            'prev_close': 146700,
            'target_price_consensus': 180000,
            'stop_loss_pct': -0.08,
            'position_weight': 0.20,
            'avg_daily_volume': 3000000,
            'war_premium_factor': 1.2,
        },
        '064350': {  # 현대로템
            'name': '현대로템',
            'prev_close': 249000,   # 추정 (8.03% 상승)
            'target_price_consensus': 310000,
            'stop_loss_pct': -0.07,
            'position_weight': 0.15,
            'avg_daily_volume': 1200000,
            'war_premium_factor': 1.15,
        },
    }
    
    def __init__(self, beta_multiplier=1.0):
        self.beta = beta_multiplier
        
    # ────────────────────────────────────────────
    # 당일 스캘핑 전략 (1분/5분봉)
    # ────────────────────────────────────────────
    
    def intraday_scalp_signal(self, code, candle_1m, candle_5m, 
                               current_price, vwap, volume_profile):
        """
        VWAP + RSI(5분) + 거래량 프로파일 기반 스캘핑
        전쟁 상황 → 변동성 극대 → 스캘핑 최적 환경
        """
        target = self.TARGETS[code]
        signal = {'action': 'HOLD', 'confidence': 0, 'size': 0}
        
        # ── 지표 계산 ──
        rsi_5m = self._calc_rsi(candle_5m['close'], period=9)
        rsi_1m = self._calc_rsi(candle_1m['close'], period=14)
        bb_upper, bb_mid, bb_lower = self._calc_bollinger(
            candle_5m['close'], period=20, std=2.0
        )
        vol_ratio = candle_1m['volume'].iloc[-1] / candle_1m['volume'].rolling(20).mean().iloc[-1]
        
        # ── 매수 조건: VWAP 지지 + RSI 과매도 반등 ──
        buy_conditions = [
            current_price > vwap * 0.995,          # VWAP 근접 또는 상회
            current_price < vwap * 1.02,           # VWAP에서 2% 이내
            rsi_5m < 40,                           # RSI 과매도 구간
            rsi_1m > rsi_1m,                       # RSI 반등 시작 (더미, 실제는 기울기)
            current_price > bb_lower,              # 볼린저 하단 위
            vol_ratio > 1.5,                       # 거래량 평균 1.5배 이상
        ]
        
        buy_score = sum(buy_conditions)
        
        # ── 매도 조건: VWAP 이탈 + RSI 과매수 ──
        sell_conditions = [
            current_price > bb_upper * 0.98,       # 볼린저 상단 근접
            rsi_5m > 75,                           # RSI 과매수
            current_price > vwap * 1.05,           # VWAP 5% 이상 괴리
            vol_ratio < 0.5,                       # 거래량 급감 (동력 소진)
        ]
        
        sell_score = sum(sell_conditions)
        
        # ── 신호 생성 ──
        if buy_score >= 4:
            signal['action'] = 'BUY'
            signal['confidence'] = min(buy_score / 6 * 100, 95)
            signal['size'] = target['position_weight'] * self.beta
            signal['entry_price'] = current_price
            signal['stop_loss'] = current_price * (1 + target['stop_loss_pct'])
            signal['take_profit_1'] = current_price * 1.03   # 1차 익절 +3%
            signal['take_profit_2'] = current_price * 1.06   # 2차 익절 +6%
            
        elif sell_score >= 3:
            signal['action'] = 'SELL'
            signal['confidence'] = min(sell_score / 4 * 100, 90)
            
        return signal
    
    # ────────────────────────────────────────────
    # 중기 스윙 전략 (일봉)
    # ────────────────────────────────────────────
    
    def swing_signal(self, code, daily_candles, war_day_count):
        """
        방산주 중기(5~20거래일) 스윙 전략
        전쟁 장기화 시나리오 기반
        
        과거 패턴:
        - 러우전쟁(2022.2): 방산주 3개월간 +80~150% 상승
        - 이스라엘-이란(2025.6): 방산주 2주간 +20~30% 상승 후 조정
        """
        target = self.TARGETS[code]
        
        # ── 일봉 지표 ──
        macd, macd_signal, macd_hist = self._calc_macd(
            daily_candles['close'], fast=12, slow=26, signal=9
        )
        adx = self._calc_adx(daily_candles, period=14)
        
        # ── 전쟁 단계별 포지션 조절 ──
        if war_day_count <= 7:
            # 전쟁 초기: 적극 매수 (모멘텀 최강 구간)
            if macd_hist.iloc[-1] > 0 and adx > 25:
                return {
                    'action': 'BUY_SWING',
                    'holding_days': 10,
                    'target': target['target_price_consensus'],
                    'stop': target['prev_close'] * 0.90,
                    'size': target['position_weight'] * self.beta * 1.2,
                }
        elif war_day_count <= 14:

    # ── 보조 지표 계산 함수 ──
    def _calc_rsi(self, prices, period=14):
        delta = prices.diff()
        gain = delta.where(delta > 0, 0).rolling(period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs)).iloc[-1]
    
    def _calc_bollinger(self, prices, period=20, std=2.0):
        mid = prices.rolling(period).mean()
        std_dev = prices.rolling(period).std()
        return mid + std * std_dev, mid, mid - std * std_dev
    
    def _calc_macd(self, prices, fast=12, slow=26, signal=9):
        ema_fast = prices.ewm(span=fast).mean()
        ema_slow = prices.ewm(span=slow).mean()
        macd = ema_fast - ema_slow
        macd_signal = macd.ewm(span=signal).mean()
        return macd, macd_signal, macd - macd_signal
    
    def _calc_adx(self, df, period=14):
        # ADX 계산 (추세 강도)
        high, low, close = df['high'], df['low'], df['close']
        tr = pd.concat([high-low, abs(high-close.shift()), abs(low-close.shift())], axis=1).max(axis=1)
        atr = tr.rolling(period).mean()
        plus_dm = (high.diff()).where(lambda x: x > 0, 0)
        minus_dm = (-low.diff()).where(lambda x: x > 0, 0)
        plus_di = 100 * (plus_dm.rolling(period).mean() / atr)
        minus_di = 100 * (minus_dm.rolling(period).mean() / atr)
        dx = 100 * abs(plus_di - minus_di) / (plus_di + minus_di)
        return dx.rolling(period).mean().iloc[-1]

            'portfolio_status': self._get_portfolio_status(),
        }
    
    def _apply_risk_filter(self, signals):
        """리스크 한도 초과 신호 필터링"""
        filtered = []
        
        for sig in signals:
            # 일일 손실 한도 체크
            if self.daily_pnl / self.total_capital < self.RISK_LIMITS['max_daily_loss']:
                continue  # 일일 한도 초과 → 신규 매수 차단
                
            # 총 낙폭 한도 체크
            current_equity = self.total_capital + self.daily_pnl
            drawdown = (current_equity - self.max_equity) / self.max_equity
            if drawdown < self.RISK_LIMITS['max_total_drawdown']:
                continue  # 총 낙폭 한도 초과 → 전면 매매 중단
                
            # 단일 종목 비중 한도
            if sig.get('size_pct', 0) > self.RISK_LIMITS['max_single_position']:
                sig['size_pct'] = self.RISK_LIMITS['max_single_position']
                
            # 최소 현금 확보
            invested_ratio = sum(p.get('weight', 0) for p in self.positions.values())
            if invested_ratio > (1 - self.RISK_LIMITS['min_cash_reserve']):
                if sig['action'] in ['BUY', 'BUY_SWING']:
                    continue  # 현금 부족 → 매수 차단
                    
            filtered.append(sig)
            
        return filtered
    
    def _analyze_news(self, news_data):
        """뉴스 키워드 기반 센티먼트 분석"""
        if not news_data:
            return 'NEUTRAL'
        text = ' '.join(news_data)
        for kw in self.NEWS_KEYWORDS['EXTREME_POSITIVE']:
            if kw in text:
                return 'EXTREME_POSITIVE'
        for kw in self.NEWS_KEYWORDS['EXTREME_NEGATIVE']:
            if kw in text:
                return 'EXTREME_NEGATIVE'
        for kw in self.NEWS_KEYWORDS['POSITIVE']:
            if kw in text:
                return 'POSITIVE'
        for kw in self.NEWS_KEYWORDS['NEGATIVE']:
            if kw in text:
                return 'NEGATIVE'
        return 'NEUTRAL'
    
    def _generate_orders(self, signals):
        """API 주문 포맷 생성 (키움/한투 호환)"""
        orders = []
        for sig in signals:
            if sig['action'] in ['BUY', 'BUY_SWING']:
                orders.append({
                    'type': 'LIMIT_BUY',
                    'code': sig.get('code', ''),
                    'price': sig.get('price', sig.get('entry_price', 0)),
                    'quantity': int(self.total_capital * sig.get('size_pct', 0.05) 
                                    / sig.get('price', sig.get('entry_price', 1))),
                    'stop_loss': sig.get('stop_loss', 0),
                    'take_profit': sig.get('take_profit_1', sig.get('take_profit', 0)),
                })
            elif sig['action'] in ['SELL', 'SELL_50PCT']:
                orders.append({
                    'type': 'MARKET_SELL',
                    'code': sig.get('code', ''),
                    'quantity': 'ALL' if sig['action'] == 'SELL' else 'HALF',
                })
        return orders
    
    def _get_portfolio_status(self):
        return {
            'total_capital': self.total_capital,
            'daily_pnl': self.daily_pnl,
            'positions': self.positions,
            'cash_ratio': 1 - sum(p.get('weight', 0) for p in self.positions.values()),
        }

class Server32Client:
    """server32 REST API 래퍼 — 키움증권 연동"""

    def __init__(self, base_url=SERVER_BASE):
        self.base = base_url
        self.account_no = None
        self.session = requests.Session()
        self.session.timeout = 10

    def _get(self, path, params=None) -> dict:
        """공통 GET 요청 — 응답 파싱"""
        try:
            r = self.session.get(f"{self.base}{path}", params=params)
            data = r.json()
            if not data.get('Success'):
                log.warning(f"API 실패: {path} → {data.get('Message')}")
            return data
        except Exception as e:
            log.error(f"API 오류: {path} → {e}")
            return {'Success': False, 'Message': str(e), 'Data': None}

    def _post(self, path, body) -> dict:
        """공통 POST 요청"""
        try:
            r = self.session.post(f"{self.base}{path}", json=body)
            return r.json()
        except Exception as e:
            log.error(f"POST 오류: {path} → {e}")
            return {'Success': False, 'Message': str(e), 'Data': None}

    # ── 인증/상태 ──
    def login(self) -> bool:
        res = self._get('/api/auth/login')
        if res['Success'] and res['Data']:
            self.account_no = res['Data'].get('account')
            log.info(f"✅ 로그인 성공: {self.account_no}")
            return True
        return False

    def check_status(self) -> dict:
        return self._get('/api/status')

    # ── 계좌 ──
    def get_dashboard(self, refresh=False) -> dict:
        path = '/api/dashboard/refresh' if refresh else '/api/dashboard'
        return self._get(path)

    def get_balance(self) -> list:
        res = self._get('/api/accounts/balance', {'accountNo': self.account_no})
        return res.get('Data', []) if res['Success'] else []

    def get_deposit(self) -> dict:
        res = self._get('/api/accounts/deposit', {'accountNo': self.account_no})
        return res.get('Data', [{}])[0] if res['Success'] and res.get('Data') else {}

    def get_outstanding_orders(self) -> list:
        res = self._get('/api/accounts/orders', {'accountNo': self.account_no})
        return res.get('Data', []) if res['Success'] else []

    # ── 시세 ──
    def get_symbol_info(self, code: str) -> dict:
        res = self._get('/api/market/symbol', {'code': code})
        return res.get('Data', {}) if res['Success'] else {}

    def get_daily_candles(self, code: str, date: str, stop_date: str) -> list:
        """일봉 조회 — date=최근, stopDate=과거"""
        res = self._get('/api/market/candles/daily', {
            'code': code, 'date': date, 'stopDate': stop_date
        })
        return res.get('Data', []) if res['Success'] else []

    def get_minute_candles(self, code: str, tick: int = 1, 
                           stop_time: str = '20180101090000') -> list:
        """분봉 조회"""
        res = self._get('/api/market/candles/minute', {
            'code': code, 'tick': str(tick), 'stopTime': stop_time
        })
        return res.get('Data', []) if res['Success'] else []

    # ── 실시간 ──
    def subscribe_realtime(self, codes: List[str], screen='1000'):
        code_str = ';'.join(codes)
        return self._get('/api/realtime/subscribe', {
            'codes': code_str, 'screen': screen
        })

    def unsubscribe_realtime(self, screen='ALL', code='ALL'):
        return self._get('/api/realtime/unsubscribe', {
            'screen': screen, 'code': code
        })

    # ── 주문 ──
    def send_order(self, code: str, order_type: int, quantity: int,
                   price: int, quote_type: str = '00') -> dict:
        """
        order_type: 1=신규매수, 2=신규매도, 3=매수취소, 4=매도취소
        quote_type: "00"=지정가, "03"=시장가
        """
        body = {
            "AccountNo": self.account_no,
            "StockCode": code,
            "OrderType": order_type,
            "Quantity": quantity,
            "Price": price,
            "QuoteType": quote_type,
        }
        log.info(f"📤 주문: {code} {'매수' if order_type==1 else '매도'} "
                 f"{quantity}주 @ {price:,}원")
        return self._post('/api/orders', body)

class MySQLClient:
    """MySQL 종목정보 + 일봉 데이터 조회/저장"""

    def __init__(self):
        self.conn_info = None
        self.conn_candle = None
        self._connect()

    def _connect(self):
        try:
            self.conn_info = mysql.connector.connect(
                **MYSQL_CONFIG, database=DB_STOCK_INFO
            )
            self.conn_candle = mysql.connector.connect(
                **MYSQL_CONFIG, database=DB_CANDLES
            )
            log.info("✅ MySQL 연결 성공")
        except Exception as e:
            log.error(f"MySQL 연결 실패: {e}")

    def get_stock_info(self, code: str) -> dict:
        """stock_base_info에서 종목 기본정보"""
        cur = self.conn_info.cursor(dictionary=True)
        cur.execute("""
            SELECT code, name, market, sector, sector_large, market_cap,
                   per, pbr, eps, bps, div_yield, foreign_ratio
            FROM stock_base_info 
            WHERE code = %s AND is_excluded = 0
        """, (code,))
        row = cur.fetchone()
        cur.close()
        return row or {}

    def get_sector_stocks(self, sector_keyword: str) -> list:
        """섹터별 종목 리스트"""
        cur = self.conn_info.cursor(dictionary=True)
        cur.execute("""
            SELECT code, name, market_cap, per, foreign_ratio
            FROM stock_base_info
            WHERE (sector LIKE %s OR sector_large LIKE %s)
              AND is_excluded = 0
            ORDER BY market_cap DESC
        """, (f'%{sector_keyword}%', f'%{sector_keyword}%'))
        rows = cur.fetchall()
        cur.close()
        return rows

    def get_daily_candles_df(self, code: str, days: int = 120) -> pd.DataFrame:
        """일봉 DataFrame — 기술적 분석용"""
        cur = self.conn_candle.cursor(dictionary=True)
        cur.execute("""
            SELECT date, open, high, low, close, volume, change_pct
            FROM daily_candles
            WHERE code = %s
            ORDER BY date DESC
            LIMIT %s
        """, (code, days))
        rows = cur.fetchall()
        cur.close()
        if not rows:
            return pd.DataFrame()
        df = pd.DataFrame(rows)
        df['date'] = pd.to_datetime(df['date'])
        df = df.sort_values('date').reset_index(drop=True)
        return df

    def save_candle(self, code: str, date: str, o: int, h: int, 
                    l: int, c: int, vol: int, chg: float = None):
        """실시간 수집된 일봉 저장/업데이트"""
        cur = self.conn_candle.cursor()
        cur.execute("""
            INSERT INTO daily_candles (code, date, open, high, low, close, volume, change_pct)
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
            ON DUPLICATE KEY UPDATE
                open=%s, high=%s, low=%s, close=%s, volume=%s, change_pct=%s
        """, (code, date, o, h, l, c, vol, chg,
              o, h, l, c, vol, chg))
        self.conn_candle.commit()
        cur.close()

# TECHNICAL ANALYSIS MODULE
# ═══════════════════════════════════════════════════════════════

class TechnicalAnalysis:
    """기술적 지표 계산 — DataFrame 기반"""

    @staticmethod
    def rsi(series: pd.Series, period: int = 14) -> pd.Series:
        delta = series.diff()
        gain = delta.where(delta > 0, 0).rolling(period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    @staticmethod
    def bollinger(series: pd.Series, period=20, std_mult=2.0):
        mid = series.rolling(period).mean()
        std = series.rolling(period).std()
        return mid + std_mult * std, mid, mid - std_mult * std

    @staticmethod
    def macd(series: pd.Series, fast=12, slow=26, signal=9):
        ema_f = series.ewm(span=fast).mean()
        ema_s = series.ewm(span=slow).mean()
        macd_line = ema_f - ema_s
        sig_line = macd_line.ewm(span=signal).mean()
        hist = macd_line - sig_line
        return macd_line, sig_line, hist

    @staticmethod
    def vwap(df: pd.DataFrame) -> pd.Series:
        """VWAP = cumsum(전형가 * 거래량) / cumsum(거래량)"""
        typical = (df['high'] + df['low'] + df['close']) / 3
        cum_tp_vol = (typical * df['volume']).cumsum()
        cum_vol = df['volume'].cumsum()
        return cum_tp_vol / cum_vol

    @staticmethod
    def adx(df: pd.DataFrame, period=14) -> pd.Series:
        high, low, close = df['high'], df['low'], df['close']
        tr = pd.concat([
            high - low,
            (high - close.shift()).abs(),
            (low - close.shift()).abs()
        ], axis=1).max(axis=1)
        atr = tr.rolling(period).mean()
        plus_dm = high.diff().clip(lower=0)
        minus_dm = (-low.diff()).clip(lower=0)
        pdi = 100 * plus_dm.rolling(period).mean() / atr
        mdi = 100 * minus_dm.rolling(period).mean() / atr
        dx = 100 * (pdi - mdi).abs() / (pdi + mdi)
        return dx.rolling(period).mean()

    @staticmethod
    def moving_averages(series: pd.Series):
        return {
            'ma5': series.rolling(5).mean(),
            'ma20': series.rolling(20).mean(),
            'ma60': series.rolling(60).mean(),
        }

    """

    Input: 코스피지수, 환율, 유가, VKOSPI, 수급, 전쟁일수
    Output: Regime, beta_multiplier (0.2 ~ 1.5)
    """
    # 과거 중동전쟁 패턴 (6건 평균)
    HISTORY = {
        'avg_1day': -0.01, 'avg_1week': +0.031, 'avg_1month': +0.046,
        'all_1month_positive': True, 'recovery_days': 20,
    }

    def __init__(self):
        self.war_start_date = datetime(2026, 2, 28)  # 이란 공습일
        self.regime = Regime.CRISIS
        self.beta = 0.5

    @property
    def war_day_count(self) -> int:
        return (datetime.now() - self.war_start_date).days

    def evaluate(self, kospi: float, usdkrw: float, wti: float,
                 vkospi: float, foreign_net: float, news_sent: str) -> Tuple[Regime, float]:
        score = 0

        # 코스피 낙폭 판단
        drop = (kospi - 6244) / 6244
        if drop < -0.10:   score -= 3
        elif drop < -0.07: score -= 1
        elif drop > -0.03: score += 2

        # 환율
        if usdkrw > 1500:   score -= 2
        elif usdkrw < 1440: score += 1

        # 유가
        if wti > 90:   score -= 3
        elif wti > 80: score -= 1
        else:          score += 1

        # VKOSPI
        if vkospi > 35:   score -= 2
        elif vkospi < 20: score += 2

        # 외국인 수급 (억원)
        if foreign_net > 10000:    score += 2
        elif foreign_net > 0:      score += 1
        elif foreign_net < -30000: score -= 2

        # 전쟁일수
        d = self.war_day_count
        if d <= 2:   score -= 2
        elif d <= 5: score -= 1
        elif d <= 10: score += 1
        else:         score += 2

        # 뉴스 센티먼트
        if news_sent == 'EXTREME_POS': score += 3
        elif news_sent == 'POS':       score += 1
        elif news_sent == 'NEG':       score -= 1
        elif news_sent == 'EXTREME_NEG': score -= 3

        # Regime 결정
        if score <= -6:
            self.regime, self.beta = Regime.EXTREME_CRISIS, 0.2
        elif score <= -3:
            self.regime, self.beta = Regime.CRISIS, 0.5
        elif score <= 0:
            self.regime, self.beta = Regime.CAUTIOUS, 0.7
        elif score <= 4:
            self.regime, self.beta = Regime.RECOVERY, 1.0
        else:
            self.regime, self.beta = Regime.AGGRESSIVE, 1.5

        log.info(f"📊 MACRO: score={score}, regime={self.regime.value}, "
                 f"beta={self.beta}, war_day={d}")

class RealtimeHandler:
    """WebSocket 실시간 데이터 수신 + 분봉 버퍼 관리"""

    def __init__(self, on_tick_callback):
        self.on_tick = on_tick_callback
        self.ws = None
        self.prices: Dict[str, int] = {}
        self.minute_buffers: Dict[str, List[dict]] = {}  # code → [{time,o,h,l,c,vol}]
        self._running = False

    def start(self):
        """WebSocket 연결 시작 (별도 스레드)"""
        self._running = True
        t = threading.Thread(target=self._connect, daemon=True)
        t.start()

    def stop(self):
        self._running = False
        if self.ws:
            self.ws.close()

    def _connect(self):
        def on_message(ws, msg):
            try:
                data = json.loads(msg)
                msg_type = data.get('type', '')
                code = data.get('code', '')

                if msg_type == 'tick' and code:
                    tick = data.get('data', {})
                    price = int(str(tick.get('current_price', '0')).lstrip('-+'))
                    volume = int(str(tick.get('volume', '0')))

                    self.prices[code] = price
                    self._update_minute_buffer(code, price, volume, tick)
                    self.on_tick(code, price, volume, tick)

            except Exception as e:
                log.debug(f"WS parse error: {e}")

        def on_error(ws, error):
            log.warning(f"WS error: {error}")

        def on_close(ws, code, reason):
            log.info(f"WS closed: {code} {reason}")
            if self._running:
                time.sleep(3)
                self._connect()

        self.ws = websocket.WebSocketApp(
            WS_REALTIME,
            on_message=on_message,
            on_error=on_error,
            on_close=on_close,
        )
        self.ws.run_forever()

    def _update_minute_buffer(self, code, price, volume, tick):
        """실시간 틱 → 1분봉 집계"""
        if code not in self.minute_buffers:
            self.minute_buffers[code] = []
        now = datetime.now()
        minute_key = now.strftime('%Y%m%d%H%M')

        buf = self.minute_buffers[code]
        if buf and buf[-1].get('_key') == minute_key:
            bar = buf[-1]
            bar['high'] = max(bar['high'], price)
            bar['low'] = min(bar['low'], price)
            bar['close'] = price
            bar['volume'] += volume
        else:
            buf.append({
                '_key': minute_key,
                'time': now,
                'open': price, 'high': price,
                'low': price, 'close': price,
                'volume': volume,
            })
            # 최근 200봉만 유지
            if len(buf) > 200:
                buf.pop(0)

    def get_minute_df(self, code: str) -> pd.DataFrame:
        """특정 종목의 분봉 DataFrame"""
        buf = self.minute_buffers.get(code, [])
        if not buf:
            return pd.DataFrame()
        df = pd.DataFrame(buf)
        return df[['time', 'open', 'high', 'low', 'close', 'volume']]

# RISK MANAGER
# ═══════════════════════════════════════════════════════════════

class RiskManager:
    """포트폴리오 레벨 리스크 관리"""

    MAX_SINGLE_POSITION = 0.15
    MAX_DAILY_LOSS = -0.05
    MAX_DRAWDOWN = -0.12
    MIN_CASH = 0.10

    def __init__(self, total_capital: int):
        self.total_capital = total_capital
        self.daily_pnl = 0
        self.max_equity = total_capital

    def can_buy(self, signal: Signal, current_holdings: list) -> bool:
        """매수 가능 여부 판단"""
        # 일일 손실 한도
        if self.daily_pnl / self.total_capital < self.MAX_DAILY_LOSS:
            log.warning(f"⛔ 일일 손실 한도 초과 → 매수 차단")
            return False

        # 낙폭 한도
        equity = self.total_capital + self.daily_pnl
        dd = (equity - self.max_equity) / self.max_equity
        if dd < self.MAX_DRAWDOWN:
            log.warning(f"⛔ 총 낙폭 한도({dd*100:.1f}%) 초과 → 전면 매매 중단")
            return False

        return True

    def adjust_quantity(self, signal: Signal) -> int:
        """리스크 기반 수량 조정"""
        max_amount = self.total_capital * self.MAX_SINGLE_POSITION
        if signal.price > 0:
            max_qty = int(max_amount / signal.price)
            return min(signal.quantity, max_qty)
        return signal.quantity

# ORDER EXECUTOR
# ═══════════════════════════════════════════════════════════════

class OrderExecutor:
    """Signal → 실제 주문 변환 + 실행"""

    def __init__(self, api: Server32Client, risk: RiskManager):
        self.api = api
        self.risk = risk
        self.pending_signals: List[Signal] = []
        self.executed_orders: List[dict] = []

    def process_signal(self, signal: Signal, holdings: list):
        """시그널을 주문으로 변환하여 실행"""
        if signal.action == 'HOLD':
            return

        signal.timestamp = datetime.now().strftime('%Y%m%d%H%M%S')

        if signal.action == 'BUY':
            if not self.risk.can_buy(signal, holdings):
                return
            qty = self.risk.adjust_quantity(signal)
            if qty <= 0:
                return

            log.info(f"🟢 매수실행: {signal.name}({signal.code}) "
                     f"{qty}주 @ {signal.price:,}원 [{signal.chart}] "
                     f"이유: {signal.reason}")

            result = self.api.send_order(
                code=signal.code,
                order_type=1,       # 신규매수
                quantity=qty,
                price=signal.price,
                quote_type='00',    # 지정가
            )
            self.executed_orders.append({
                'signal': signal, 'result': result, 'time': signal.timestamp
            })

        elif signal.action == 'SELL':
            # 보유 종목에서 수량 조회
            held_qty = 0
            for h in holdings:
                if h.get('종목코드', '').strip() == signal.code:
                    held_qty = int(h.get('보유수량', '0'))
                    break

            sell_qty = held_qty if signal.quantity == 0 else min(signal.quantity, held_qty)
            if sell_qty <= 0:
                return

            log.info(f"🔴 매도실행: {signal.name}({signal.code}) "
                     f"{sell_qty}주 @ 시장가 [{signal.chart}] "
                     f"이유: {signal.reason}")

            result = self.api.send_order(
                code=signal.code,
                order_type=2,       # 신규매도
                quantity=sell_qty,
                price=0,
                quote_type='03',    # 시장가
            )
            self.executed_orders.append({
                'signal': signal, 'result': result, 'time': signal.timestamp
            })

                # ── 계좌 잔고 조회 ──
                holdings = self.api.get_balance()

                # ── CHART 2~4: 종목별 시그널 생성 ──
                all_signals: List[Signal] = []

                for code, target in TARGETS.items():
                    price = self.rt.prices.get(code, target.prev_close)
                    if price <= 0:
                        continue

                    # 분봉 데이터
                    min_df = self.rt.get_minute_df(code)
                    # 분봉이 부족하면 API에서 가져오기
                    if len(min_df) < 20:
                        raw = self.api.get_minute_candles(code, tick=5)
                        if raw:
                            min_df = self._parse_minute_candles(raw)

                    # 일봉 데이터 (MySQL)
                    daily_df = self.db.get_daily_candles_df(code, days=60)

                    # ── Chart 2: 방산 ──
                    if target.sector == 'defense':
                        sig = self.chart2.generate_signal(
                            code, min_df, daily_df, beta, price
                        )
                        if sig:
                            all_signals.append(sig)

                    # ── Chart 3: 에너지 ──
                    elif target.sector == 'energy':
                        sig = self.chart3.generate_signal(
                            code, min_df, self.ext.latest_wti,
                            self.ext.wti_change_pct, beta, price,
                            self.ext.news_sentiment,
                        )
                        if sig:
                            all_signals.append(sig)

                    # ── Chart 4: 반도체 ──
                    elif target.sector == 'semiconductor':
                        sigs = self.chart4.generate_signals(
                            code, daily_df, price, beta,
                            self.chart1.war_day_count,
                            self.ext.nvidia_change,
                        )
                        all_signals.extend(sigs)

                # ── 주문 실행 ──
                for sig in all_signals:
                    self.executor.process_signal(sig, holdings)

                # ── 사이클 로깅 ──
                elapsed = time.time() - cycle_start
                if self._cycle_count % 5 == 0:  # 5분마다 요약
                    log.info(
                        f"─── Cycle #{self._cycle_count} ({elapsed:.1f}s) ───\n"
                        f"  Regime: {regime.value} | Beta: {beta:.2f}\n"
                        f"  WTI: ${self.ext.latest_wti:.1f} ({self.ext.wti_change_pct*100:+.1f}%)\n"

    def _parse_minute_candles(self, raw: list) -> pd.DataFrame:
        """API 분봉 응답 → DataFrame 변환"""
        rows = []
        for r in raw:
            rows.append({
                'time': pd.to_datetime(str(r.get('체결시간', '')), format='%Y%m%d%H%M%S', errors='coerce'),
                'open': abs(int(r.get('시가', 0))),
                'high': abs(int(r.get('고가', 0))),
                'low': abs(int(r.get('저가', 0))),
                'close': abs(int(r.get('현재가', 0))),
                'volume': abs(int(r.get('거래량', 0))),
            })
        df = pd.DataFrame(rows)
        if not df.empty:
            df = df.sort_values('time').reset_index(drop=True)
        return df

        Try

                                 Dim buffer(4096) As Byte
                                 While _wsRealtime.State = WebSocketState.Open
                                     Dim result = Await _wsRealtime.ReceiveAsync(
                                         New ArraySegment(Of Byte)(buffer), CancellationToken.None)
                                     If result.MessageType = WebSocketMessageType.Text Then
                                         Dim msg = Encoding.UTF8.GetString(buffer, 0, result.Count)
                                         ParseRealtimeMessage(msg)

    Private Sub ParseRealtimeMessage(msg As String)
        Try
            Dim json = JObject.Parse(msg)
            Dim msgType = json("type")?.ToString()
            Dim code = json("code")?.ToString()

            If msgType = "tick" AndAlso code IsNot Nothing Then
                Dim price = Math.Abs(CInt(json("data")?("current_price")?.ToString() ?? "0"))
                If price > 0 Then
                    SyncLock _prices
                        _prices(code) = price
                    End SyncLock
                End If
            End If
        Catch
        End Try
    End Sub
