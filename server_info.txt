 KIWOOM / CYBOS FULL MANUAL (CONSOLE)
================================================================================================================
Base URL            : http://localhost:8082
Realtime WS         : ws://localhost:8082/ws/realtime
Execution WS        : ws://localhost:8082/ws/execution
Interactive Help    : http://localhost:8082/help
Machine JSON Manual : http://localhost:8082/api/help

[0] Global Rules
  - Every REST response envelope:
    { "Success": bool, "Message": string, "Data": object|array|null }
  - Parser order: Success -> Message -> Data
  - If Success=false: treat Message as error reason; Data may be null
  - Key principle: implement parser by endpoint Data contract below

[1] WebSocket Contracts
  A) Realtime stream
     connect: ws://localhost:8082/ws/realtime
     subscribe trigger: http://localhost:8082/api/realtime/subscribe?codes=005930;000660&screen=1000
     payload: { "type": "tick|hoga|condition", "code": string, "timestamp": "yyyyMMddHHmmss", "data": object }
     tick data typical keys: time,current_price,diff,rate,volume,cum_volume,open,high,low,intensity
     hoga data typical keys: total_ask_vol,total_bid_vol,ask_price_1..5,ask_vol_1..5,bid_price_1..5,bid_vol_1..5
     condition data keys    : condition_name, condition_index, state(enter|exit)
  B) Execution stream
     connect: ws://localhost:8082/ws/execution
     payload: { "type": "order|balance|dashboard", "timestamp": "yyyyMMddHHmmss", "data": object }
     order/balance data: chejan fid-key dictionary strings
     dashboard data   : AccountSnapshot object
     stop realtime reg: http://localhost:8082/api/realtime/unsubscribe?screen=1000&code=ALL

    Private ReadOnly _cpSvr8561T As New DSCBO1Lib.CpSvr8561T()
    Private ReadOnly _cpSvr8563 As New CpSvr8563() ' Used to get a list of themes
    Public Function GetAllThemes() As List(Of ThemeInfo)

        ' HTS와 동일하게 당일 상승률 상위 테마를 가져오도록 수정
        _cpSvr8563.SetInputValue(0, "1"c) ' '1': 전일대비상승율상위순
        _cpSvr8563.BlockRequest()

        Dim themes As New List(Of ThemeInfo)
        Dim themeCount = _cpSvr8563.GetHeaderValue(0) ' Header 0: data 건수

        For i = 0 To themeCount - 1
            themes.Add(New ThemeInfo With {
                .ThemeID = _cpSvr8563.GetDataValue(0, i).ToString(),
                .ThemeName = _cpSvr8563.GetDataValue(1, i).ToString(),
                .AvgChangeRate = Convert.ToDouble(_cpSvr8563.GetDataValue(3, i)) ' 3: 1일전대비 (당일 등락률)
            })
        Next
        Return themes
    End Function

    Public Function GetComponentsForTheme(themeCode As String) As List(Of SectorComponent)
        EnsureConnected()
        Util.LogToFile($"DIAGNOSTIC: Entering GetComponentsForTheme for {themeCode} on Thread ID: {Thread.CurrentThread.ManagedThreadId}")
        Dim sw As New Stopwatch()
        sw.Start()
        _cpSvr8561T.SetInputValue(0, CShort(themeCode))
        _cpSvr8561T.BlockRequest()
        sw.Stop()
        Util.LogToFile($"DIAGNOSTIC: GetComponentsForTheme BlockRequest() finished in {sw.ElapsedMilliseconds}ms on Thread ID: {Thread.CurrentThread.ManagedThreadId}")

        Dim components As New List(Of SectorComponent)
        Dim stockCount = _cpSvr8561T.GetHeaderValue(1)
        If stockCount = 0 Then Return components

        For i = 0 To stockCount - 1
            ' Data based on 테마별종목.txt spec
            Dim code = _cpSvr8561T.GetDataValue(0, i).ToString().Trim()
            Dim name = _cpSvr8561T.GetDataValue(1, i).ToString().Trim()
            Dim price = Util.sTol(_cpSvr8561T.GetDataValue(2, i).ToString()) ' Util 함수 사용
            Dim changeRate = Util.sTod(_cpSvr8561T.GetDataValue(4, i).ToString(), True) ' 등락률 (부호 유지)
            Dim volume = Util.sTol(_cpSvr8561T.GetDataValue(5, i).ToString()) ' 거래량

            ' Calculate Trading Value in millions
            Dim tradingValue = 0L
            If price > 0 AndAlso volume > 0 Then
                tradingValue = Convert.ToInt64((CDec(price) * CDec(volume)) / 1000000D)
            End If

            components.Add(New SectorComponent With {
                .StockCode = code.TrimStart("A"c), ' Normalize code for Kiwoom API (Remove 'A' prefix)
                .StockName = name,
                .CurrentPrice = price,
                .ChangeRate = changeRate,
                .Volume = volume,
                .TradingValue = tradingValue
            })
        Next
        Return components
    End Function

    ''' <summary>
    ''' 지정된 과거 날짜를 기준으로 각 테마의 평균 등락률을 계산하여 테마 목록을 반환합니다.
    ''' 이 함수는 많은 데이터를 요청하므로 시간이 걸릴 수 있습니다.
    ''' </summary>
    ''' <param name="targetDate">분석할 과거 날짜입니다.</param>
    ''' <returns>과거 등락률이 포함된 ThemeInfo 리스트입니다.</returns>
    Public Async Function GetAllThemesWithPastChangeRate(targetDate As DateTime) As Task(Of List(Of ThemeInfo))
        EnsureConnected()

        ' 1. 현재 기준의 모든 테마 목록을 가져옵니다.
        Dim allThemes = GetAllThemes()
        Debug.Print($"총 {allThemes.Count}개의 테마 목록 확인. 과거 등락률 재구성 시작...")

        ' 2. 분석일과 그 전 거래일을 yyyyMMdd 형식으로 준비합니다.
        Dim targetDateStr = targetDate.ToString("yyyyMMdd")
        Dim prevDate = targetDate.AddDays(-1)
        ' 주말/휴일 처리
        While prevDate.DayOfWeek = DayOfWeek.Saturday OrElse prevDate.DayOfWeek = DayOfWeek.Sunday
            prevDate = prevDate.AddDays(-1)
        End While
        Dim prevDateStr = prevDate.ToString("yyyyMMdd")

        ' 3. 각 테마의 평균 등락률을 계산합니다.
        For Each theme In allThemes
            Dim components = GetComponentsForTheme(theme.ThemeID)
            If Not components.Any() Then Continue For

            Dim totalChangeRate As Double = 0
            Dim validComponentCount As Integer = 0

            For Each component In components
                Try
                    ' 해당 종목의 이틀치 일봉 데이터를 가져옵니다.
                    Dim candles = Await getCandlesWithPeriodAsync(component.StockCode, "D1", prevDateStr, targetDateStr)
                    If candles Is Nothing OrElse candles.Count < 2 Then Continue For

                    Dim prevDayCandle = candles.FirstOrDefault(Function(c) c.Timestamp.ToString("yyyyMMdd") = prevDateStr)
                    Dim targetDayCandle = candles.FirstOrDefault(Function(c) c.Timestamp.ToString("yyyyMMdd") = targetDateStr)

                    If prevDayCandle IsNot Nothing AndAlso targetDayCandle IsNot Nothing AndAlso prevDayCandle.Close > 0 Then
                        Dim changeRate = (targetDayCandle.Close - prevDayCandle.Close) / prevDayCandle.Close * 100
                        totalChangeRate += changeRate
                        validComponentCount += 1
                    End If

                    Await Task.Delay(250) ' Cybos API 요청 제한 준수 (1초에 4회)
                Catch ex As Exception
                    Debug.Print($"종목({component.StockCode}) 과거 데이터 조회 중 오류: {ex.Message}")
                End Try
            Next

            If validComponentCount > 0 Then
                theme.AvgChangeRate = totalChangeRate / validComponentCount
            Else
                theme.AvgChangeRate = 0
            End If
            Debug.Print($"테마 '{theme.ThemeName}' 분석 완료. 평균 등락률: {theme.AvgChangeRate:F2}%")
        Next

        ' 4. 계산된 평균 등락률로 테마 목록을 정렬하여 반환합니다.
        Return allThemes.OrderByDescending(Function(t) t.AvgChangeRate).ToList()
    End Function

Public Class FastGrid

    End Class
    Private _columns As New List(Of GridColumn)()
    Private _rows As New List(Of Dictionary(Of String, String))()

    ' Events
    Public Event CellDoubleClick As EventHandler(Of DataGridViewCellEventArgs)
    Public Event CellClick As EventHandler(Of DataGridViewCellEventArgs)
    Public Event CellValueNeeded As EventHandler(Of CellValueNeededEventArgs)

    Public Sub New()
        Me.DoubleBuffered = True
        Me.ResizeRedraw = True
        
        _vScrollBar = New VScrollBar()
        _vScrollBar.Dock = DockStyle.Right
        _vScrollBar.Visible = False
        Me.Controls.Add(_vScrollBar)
        
        AddHandler _vScrollBar.Scroll, AddressOf OnScroll
    End Sub

    ' ==========================================
    ' Public API (Simulator Compatibility)
    ' ==========================================
    Public Sub AddColumn(name As String, title As String, width As Integer)
        _columns.Add(New GridColumn With {.Name = name, .Title = title, .Width = width})
        Me.Invalidate()
    End Sub

    Public Function AddRow() As Dictionary(Of String, String)
        Dim row As New Dictionary(Of String, String)()
        _rows.Add(row)
        UpdateScroll()
        Me.Invalidate()
        Return row
    End Function

    Public Sub ClearRows()
        _rows.Clear()
        UpdateScroll()
        Me.Invalidate()
    End Sub

        Set(value As Integer)
            ' 컬럼 개수 맞추기 (단순 리사이즈)
            While _columns.Count < value
                _columns.Add(New GridColumn With {.Name = "Col" & _columns.Count, .Title = "Col" & _columns.Count, .Width = _legacyColumnWidth})
            End While
            If _columns.Count > value Then
                _columns.RemoveRange(value, _columns.Count - value)
            End If
            Me.Invalidate()
        End Set
    End Property

    <DesignerSerializationVisibility(DesignerSerializationVisibility.Hidden)>
    Public Property ColumnTitles As String()
        Get
            Return _columns.Select(Function(c) c.Title).ToArray()
        End Get
        Set(value As String())
            If value Is Nothing Then Return
            ' 기존 컬럼 타이틀 업데이트 또는 새 컬럼 추가
            For i = 0 To value.Length - 1
                If i < _columns.Count Then
                    _columns(i).Title = value(i)
                Else
                    _columns.Add(New GridColumn With {.Name = "Col" & i, .Title = value(i), .Width = _legacyColumnWidth})
                End If
            Next
            Me.Invalidate()
        End Set
    End Property

    ' ==========================================
    ' Shared RowCount
    ' ==========================================
    Private _virtualRowCount As Integer = 0
    
    <DesignerSerializationVisibility(DesignerSerializationVisibility.Hidden)>
    Public Property RowCount As Integer
        Get
            Return Math.Max(_rows.Count, _virtualRowCount)
        End Get
        Set(value As Integer)
            _virtualRowCount = value
            UpdateScroll()
            Me.Invalidate()
        End Set
    End Property

    ' ==========================================
    ' Internal Logic
    ' ==========================================
    Protected Overrides Sub OnMouseDoubleClick(e As MouseEventArgs)
        MyBase.OnMouseDoubleClick(e)
        Dim y = e.Y - _headerHeight
        If y < 0 Then Return

        Dim clickedRowIndex = _vScrollBar.Value + (y \ _rowHeight)
        If clickedRowIndex >= 0 AndAlso clickedRowIndex < Me.RowCount Then
            RaiseEvent CellDoubleClick(Me, New DataGridViewCellEventArgs(0, clickedRowIndex))
        End If
    End Sub

    Protected Overrides Sub OnMouseClick(e As MouseEventArgs)
        MyBase.OnMouseClick(e)
        Dim y = e.Y - _headerHeight
        If y < 0 Then Return

        Dim clickedRowIndex = _vScrollBar.Value + (y \ _rowHeight)
        If clickedRowIndex >= 0 AndAlso clickedRowIndex < Me.RowCount Then
            RaiseEvent CellClick(Me, New DataGridViewCellEventArgs(0, clickedRowIndex))
        End If
    End Sub

    Private Sub UpdateScroll()
        Dim totalRows = Me.RowCount
        Dim visibleRows = GetVisibleRowCount()
        If totalRows > visibleRows Then
            _vScrollBar.Visible = True
            _vScrollBar.Maximum = totalRows - 1
            _vScrollBar.LargeChange = visibleRows
        Else
            _vScrollBar.Visible = False
            _vScrollBar.Value = 0
        End If
    End Sub

    Private Function GetVisibleRowCount() As Integer
        Return Math.Max(1, (Me.Height - _headerHeight) \ _rowHeight)
    End Function

    Private Sub OnScroll(sender As Object, e As ScrollEventArgs)
        Me.Invalidate()
    End Sub

    Protected Overrides Sub OnResize(e As EventArgs)
        MyBase.OnResize(e)
        UpdateScroll()
    End Sub

    Protected Overrides Sub OnMouseWheel(e As MouseEventArgs)
        If _vScrollBar.Visible Then
            Dim newValue = _vScrollBar.Value - (e.Delta \ 120) * 3
            newValue = Math.Max(0, Math.Min(newValue, _vScrollBar.Maximum - _vScrollBar.LargeChange + 1))
            _vScrollBar.Value = newValue
            Me.Invalidate()
        End If
        MyBase.OnMouseWheel(e)
    End Sub

    Protected Overrides Sub OnPaint(e As PaintEventArgs)
        MyBase.OnPaint(e)
        Dim g = e.Graphics
        
        ' 1. Draw Header
        Dim headerRect As New Rectangle(0, 0, Me.Width, _headerHeight)
        g.FillRectangle(New SolidBrush(Color.FromArgb(40, 40, 40)), headerRect)
        g.DrawLine(Pens.Gray, 0, _headerHeight - 1, Me.Width, _headerHeight - 1)

        Dim hX As Integer = 0
        ' Use defined columns or default if empty
        Dim drawCols As List(Of GridColumn) = _columns
        
        For Each col In drawCols
            Dim cellRect As New Rectangle(hX, 0, col.Width, _headerHeight)
            TextRenderer.DrawText(g, col.Title, Me.Font, cellRect, Color.Silver, TextFormatFlags.VerticalCenter Or TextFormatFlags.HorizontalCenter)
            g.DrawLine(Pens.DimGray, cellRect.Right, 0, cellRect.Right, _headerHeight)
            hX += col.Width
        Next

        ' 2. Draw Data Rows
        Dim scrollOffset = _vScrollBar.Value
        Dim visibleRows = GetVisibleRowCount() + 1
        Dim totalRows = Me.RowCount
        
        Dim firstRow = scrollOffset
        Dim lastRow = Math.Min(totalRows - 1, firstRow + visibleRows)

        For i = firstRow To lastRow
            Dim y = _headerHeight + (i - firstRow) * _rowHeight
            Dim rowRect As New Rectangle(0, y, Me.Width - (If(_vScrollBar.Visible, _vScrollBar.Width, 0)), _rowHeight)
            
            ' Background
            If i Mod 2 = 0 Then
                g.FillRectangle(Brushes.Black, rowRect)
            Else
                Using b As New SolidBrush(Color.FromArgb(20, 20, 20))
                    g.FillRectangle(b, rowRect)
                End Using
            End If
            
            ' Cell Values
            Dim x As Integer = 0
            
            For c = 0 To drawCols.Count - 1
                Dim col = drawCols(c)
                Dim cellRect As New Rectangle(x, y, col.Width, _rowHeight)
                
                Dim val As String = ""
                Dim textColor As Color = Color.White
                
                ' Priority 1: Internal Rows (Simulator)
                If i < _rows.Count AndAlso _rows(i).ContainsKey(col.Name) Then
                    val = _rows(i)(col.Name)
                    
                     ' Simple Coloring for Simulator
                    If col.Name = "PnL" OrElse col.Name = "HighPnL" Then
                        If val.StartsWith("-") Then textColor = Color.DodgerBlue Else textColor = Color.Crimson
                    ElseIf col.Name = "Score" Then
                         Dim score As Double
                         If Double.TryParse(val, score) AndAlso score >= 80 Then textColor = Color.Gold
                    End If
                Else
                    ' Priority 2: Virtual Mode Event (MainApp)
                    Dim args As New CellValueNeededEventArgs(i, c)
                    RaiseEvent CellValueNeeded(Me, args)
                    If args.Value IsNot Nothing Then val = args.Value.ToString()
                    If Not args.TextColor.IsEmpty Then textColor = args.TextColor
                End If

                TextRenderer.DrawText(g, val, Me.Font, cellRect, textColor, TextFormatFlags.VerticalCenter Or TextFormatFlags.Right)
                g.DrawLine(Pens.DimGray, cellRect.Right, cellRect.Top, cellRect.Right, cellRect.Bottom)
                x += col.Width
            Next
            
            g.DrawLine(Pens.DimGray, rowRect.Left, rowRect.Bottom, rowRect.Right, rowRect.Bottom)
        Next
    End Sub

End Class

Public Class CellValueNeededEventArgs
    Inherits EventArgs
    Public Property RowIndex As Integer
    Public Property ColumnIndex As Integer
    Public Property Value As Object
    Public Property TextColor As Color 

    Public Sub New(r As Integer, c As Integer)
        RowIndex = r
        ColumnIndex = c
        TextColor = Color.Empty 
    End Sub
End Class
